Body.jsx
import { useState } from 'react'; //useState로 만든 변수는 값이 변하면 리액트가 그걸 알아채고 화면을 자동으로 새로고침해준다
import './Body.css';

export default function Body() {
    const quotes = [
        "가장 아름다운 경치는 가장 힘든 등반 후에 온다.",
        "성공이 행복의 열쇠가 아니라, 행복이 성공의 열쇠다. ",
        "할 수 있다고 믿으면 이미 반쯤 도착한 것이다.",
        "더 이상 사랑하지 않을 때까지 당신이 하는 일을 사랑하라.",
        "최고의 복수는 무시하고 성공하는 것이다.",
        "만족할 줄 아는 것이야말로 가장 큰 넉넉함이다. ",
        "생각하는 대로 자신이 된다.",
        "오늘의 우리는 과거의 생각한 모습이다.",
        "마음속에 화를 품고 있는 것은 자신이 독약을 마시고 상대방이 죽기를 바라는 것과 같다.",
        "하루에 3시간을 걸으면 7년 후에 지구를 한 바퀴 돌 수 있다.",
        "잘하기 위해서 시작해야 하며, 시작할 때부터 잘할 필요는 없다.",
        "에너지와 끈기는 모든 것을 이겨낸다.",
    ]
    const [currentQuote, setCurrentQuote] = useState(quotes[1]) // currentQuote (데이터 부르기): * 현재 화면에 보여지고 있는 "상태 값"
                                                                // setCurrentQuote (데이터 바꾸기): * 이 값을 바꾸고 싶을 때 사용하는 전용 함수- 새명언을 불러올 때 전용함수
    const changeQuote = () => {
        const randomIndex = Math.floor(Math.random() * quotes.length);
        setCurrentQuote(quotes[randomIndex]);
    }
    return (
        <div className="body-container">
            <h1 className="body-title">오늘의 명언 제조기</h1>
            
            <div className="quote-box">
                <p className="quote-text">"{currentQuote}"</p>
            </div>

            <button className="change-btn" onClick={changeQuote}>
                새로운 명언 보기
            </button>
        </div>
    );
}
